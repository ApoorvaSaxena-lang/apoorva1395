import 'package:flutter/material.dart';

import 'profile_screen.dart';

class DashboardScreen extends StatefulWidget {
  const DashboardScreen({Key? key}) : super(key: key);

  @override
  State<DashboardScreen> createState() => _DashboardScreenState();
}

class _DashboardScreenState extends State<DashboardScreen> {
  @override
  Widget build(BuildContext context) {
    final size = MediaQuery.of(context).size / 100;
    return Scaffold(
      backgroundColor: Colors.white,
      appBar: AppBar(
        backgroundColor: Colors.white,
        elevation: 0,
        title: Text(
          "Dashboard",
          style: TextStyle(
              color: Colors.black,
              fontSize: size.height * 2.5,
              fontWeight: FontWeight.w500),
        ),
        actions: [
          Padding(
            padding: EdgeInsets.only(right: size.width * 2.5),
            child: IconButton(
                onPressed: () {
                  Navigator.push(
                      context,
                      MaterialPageRoute(
                        builder: (context) => const ProfileScreen(),
                      ));
                },
                icon: CircleAvatar(
                  radius: size.height * 1.8,
                  backgroundColor: Color(0xFFC0C0C0),
                  child: const Icon(
                    Icons.person,
                    color: Colors.white,
                  ),
                )),
          )
        ],
      ),
      body: Column(
        children: [
          Container(
              height: size.height * 15,
              decoration: BoxDecoration(
                  boxShadow: const [
                    BoxShadow(
                        color: Color.fromRGBO(0, 0, 0, 0.6000000238418579),
                        offset: Offset(0, 8),
                        blurRadius: 20)
                  ],
                  color: Color(0xFF0166C4),
                  border: Border.all(
                    color: Color.fromRGBO(1, 102, 196, 1),
                    width: 3,
                  ),
                  shape: BoxShape.circle)),
        ],
      ),
    );
  }
}
